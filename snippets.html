Request input (number)

[.js]

Requests input from user and runs a number check.

function set(userPrompt,errorMsg) {
    let inputVar = 0;
    let ok = false;
    do {
        inputVar = parseInt(prompt(userPrompt));
        if (!(isNaN(inputVar))&&(inputVar>1)) {
            ok = true;
        } else {
            alert(errorMsg);
        }
    } while (ok == false)
    return inputVar;
}

Generate random number

[.js]

Generates random number between n1 and n2 (inclusive).

function random(n1, n2) {
    let range = n2-n1+1;
    return n1+Math.floor(Math.random()*range);
}
